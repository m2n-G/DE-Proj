               |   SELECT stock_code, signal, open, MA_S, MA_L, ma_combo, date
               |   FROM signal_history
               |   WHERE date = '2024-01-15'
               |     AND signal = 'golden_cross'
               |   ORDER BY date;
               |
               |   -- 신호 발생 후 종가 기준 수익 여부 사후 검증
               |   SELECT s.stock_code,
               |          s.open        AS signal_price,
               |          d.close       AS close_price,
               |          ROUND((d.close - s.open) / s.open * 100, 2) AS return_pct
               |   FROM signal_history s
               |   JOIN daily_ohlcv d
               |     ON s.stock_code = d.stock_code AND s.date = d.date
               |   WHERE s.date = '2024-01-15';
